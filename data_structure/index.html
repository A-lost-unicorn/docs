                                # 直接插入排序

                                void InsertSort(int a[],int n){
    int i , j , temp ;
    for(i=1 ; i<n ; i++){
        if(a[i]<a[i-1]){
            temp = a[i];
            for(j=i-1 ; j>=0&&a[j]>temp ; j--){
                a[j+1] = a[j];
            }
            a[j+1] = temp;
        }
    }
}
